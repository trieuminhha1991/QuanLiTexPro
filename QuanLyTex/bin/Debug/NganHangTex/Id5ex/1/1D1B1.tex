\begin{ex}%[Thi thử, Sở GD và ĐT - Hà Tĩnh, 2019]%[Đặng Tân Hoài, 12-EX-5-2019]%[1D1B1-5]
	Gọi $M$, $m$ lần lượt là giá trị lớn nhất và giá trị nhỏ nhất của hàm số $y=1-\sin x$. Khẳng định nào sau đây đúng?
	\choice
	{\True $M=2,~m=0$}
	{$M=1,~m=-1$}
	{$M=2,~m=-1$}
	{$M=1,~m=0$}
	\loigiai{
	Với mọi $x \in \mathbb{R}$, ta có $-1 \le \sin x \le 1 \Leftrightarrow 1 \ge -\sin x \ge -1 \Leftrightarrow 2 \ge 1-\sin x \ge 0 \Leftrightarrow 2 \ge y \ge 0$.\\
	Vậy $M=\max\limits_{x \in \mathbb{R}}y=2$ đạt tại $x=-\dfrac{\pi}{2}+k2\pi,~k \in \mathbb{Z}$; $m=\min\limits_{x \in \mathbb{R}}y=0$ đạt tại $x=\dfrac{\pi}{2}+k2\pi,~k \in \mathbb{Z}$.
	}
\end{ex}
