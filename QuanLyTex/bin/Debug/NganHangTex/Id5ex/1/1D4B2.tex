\begin{ex}%[KSCL, Sở GD và ĐT - Thanh Hóa, 2018]%[Bùi Ngọc Diệp, 12EX-5]%[1D4B2-7]
	Tính giới hạn $\lim \dfrac{4n + 2018}{2n+1}$.
	\choice
	{4}
	{\True 2}
	{2018}
	{$\dfrac{1}{2}$}
	\loigiai{
		$\lim \dfrac{4n + 2018}{2n+1} = \lim \dfrac{4 + \dfrac{2018}{n}}{2 + \dfrac{1}{n}} = \dfrac{4 + 0}{2 + 0} = 2$. }
\end{ex}%!Cau!%
\begin{ex}%[Đề tập huấn, Sở GD và ĐT - Vĩnh Phúc, 2019]%[Mai Sương, EX-5-2019]%[1D4B2-7]
$\lim\limits_{x\to + \infty}\dfrac{x - 2}{x + 3}$ bằng
	\choice
	{$-\dfrac{2}{3}$}
	{\True $1$}
	{$2$}
	{$-3$}
	\loigiai{
	Ta có $\lim\limits_{x\to + \infty}\dfrac{x - 2}{x + 3}=\lim\limits_{x\to + \infty}\dfrac{1 - \dfrac{2}{x}}{1 + \dfrac{3}{x}}= 1.$	
	}
	
	\end{ex}%!Cau!%
\begin{ex}%[Thi thử, Sở GD và ĐT - Hà Tĩnh, 2019]%[Đặng Tân Hoài, 12-EX-5-2019]%[1D4B2-7]
	$\lim\limits_{x \to +\infty} \dfrac{2x-6}{x+2}$ bằng
	\choice
	{\True $ 2 $}
	{$ -2 $}
	{$ 3 $}
	{$ -3 $}
	\loigiai{
	Ta có $\lim\limits_{x \to +\infty} \dfrac{2x-6}{x+2} = \lim\limits_{x \to +\infty} \dfrac{2-\frac{6}{x}}{1+\frac{2}{x}}=\dfrac{2-0}{1+0}=2$.
	}
\end{ex}%!Cau!%
\begin{ex}%[Thử sức trước kì thi đề số 4, THTT, 2019]%[Trần Hòa, 12EX-7-2019]%[1D4B2-3]
	Giới hạn $\lim\limits_{x\to 3}\dfrac{x+1-\sqrt{5x+1}}{x-\sqrt{4x-3}}$ bằng $\dfrac{a}{b}$ (phân số tối giản, $a>0$). Giá trị của $a-b$ là
	\choice
	{\True $1$}
	{$\dfrac{1}{9}$}
	{$-1$}
	{$\dfrac{9}{8}$}
	\loigiai{
		{\allowdisplaybreaks
	\begin{eqnarray*}
	& &\lim\limits_{x\to 3}\dfrac{x+1-\sqrt{5x+1}}{x-\sqrt{4x-3}}\\
	&=&\lim\limits_{x\to 3}\dfrac{(x^2-3x)(x+\sqrt{4x-3})}{(x-1)(x-3)(x+1+\sqrt{5x+1})}\\
	&=&\lim\limits_{x\to 3}\dfrac{x(x+\sqrt{4x-3})}{(x-1)(x+1+\sqrt{5x+1})}\\
	&=&\dfrac{9}{8}.
	\end{eqnarray*}
	}
	Suy ra $a=9$, $b=8$. Vậy $a-b=1.$		
	}
\end{ex}%!Cau!%
\begin{ex}%[Thi thử, Sở GD và ĐT - Vĩnh Phúc, 2019]%[Nguyễn Văn Nay, 12EX9]%[1D4B2-3]
	Tính $\lim \limits_{x\rightarrow 1}\dfrac{x^2 - (a + 2) x + a + 1}{x^3 - 1}$
	\choice
	{$\dfrac{2 - a}{3}$}
	{$\dfrac{ - 2 - a}{3}$}
	{\True $\dfrac{ - a}{3}$}
	{$\dfrac{a}{3}$}
	\loigiai{
	Ta có
	\begin{eqnarray*}
		&&\lim \limits_{x\rightarrow 1}\dfrac{x^2 - (a + 2) x + a + 1}{x^3 - 1}\\
		&=&\lim \limits_{x\rightarrow 1}\dfrac{(x-1)(x-a-1)}{(x-1)(x^2+x+1)}\\
		&=&\lim \limits_{x\rightarrow 1}\dfrac{x-a-1}{x^2+x+1}=\dfrac{-a}{3}.
	\end{eqnarray*}
	}
\end{ex}
