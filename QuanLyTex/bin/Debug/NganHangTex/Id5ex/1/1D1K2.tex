	Nghiệm của phương trình $\sin 3x = \cos x$ là 
	\choice
	{$x = \pm \dfrac{\pi}{4} + k2\pi, k \in \Bbb{Z}$}
	{$x = \dfrac{\pi}{4} - k\pi, k \in \Bbb{Z}$}
	{\True $x = \dfrac{\pi}{8} + \dfrac{k\pi}{2}, x= \dfrac{\pi}{4} + k \pi,  k \in \Bbb{Z}$}
	{$x = \dfrac{\pi}{2} + k\pi, k \in \Bbb{Z}$}
	\loigiai{
		$\sin 3x = \cos x \Leftrightarrow \sin 3x = \sin \left(\dfrac{\pi}{2} - x\right) \Leftrightarrow \hoac{& 3x = \dfrac{\pi}{2} - x + k 2\pi\\& 3x =\dfrac{\pi}{2} + x+ k 2\pi} \Leftrightarrow \hoac{& x = \dfrac{\pi}{8} +\dfrac{k\pi}{2}\\ &x = \dfrac{\pi}{4} + k\pi} (k \in \Bbb{Z}).$
	}
\end{ex}
