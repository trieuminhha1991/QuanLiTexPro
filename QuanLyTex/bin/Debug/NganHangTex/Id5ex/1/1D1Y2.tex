\begin{ex}%[Đề thi thử THPTQG lần 2 THPT Thoại Ngọc Hầu, An Giang, năm 2019]%[Nguyễn Thành Khang, dự án 2019-Ex-7]%[1D1Y2-1]
	Phương trình $\cos2x=m$ vô nghiệm khi
	\choice
	{$m<-1$}
	{$m>1$}
	{$-1\le m\le 1$}
	{\True $\hoac{&m<-1 \\ &m>1}$}
	\loigiai{
		Vì $|\cos 2x|\le 1,\ \forall x$ nên phương trình $\cos2x=m$ vô nghiệm khi và chỉ khi $\hoac{&m<-1 \\ &m>1.}$
	}
\end{ex}%!Cau!%
\begin{ex}%[KSKT lần 2, Sở Vĩnh Phúc, 2019 ]%[Nguyễn Thế Anh, 12EX9-2019]%[1D1Y2-1]
Nghiệm của phương trình $\cot 3x=-1$ là
\choice
{$x=\dfrac{\pi}{12}+k\dfrac{\pi}{3}$ với $k\in\mathbb{Z}$}
{$x=-\dfrac{\pi}{12}+k\pi$  với $k\in\mathbb{Z}$}
{\True $x=-\dfrac{\pi}{12}+k\dfrac{\pi}{3}$ với $k\in\mathbb{Z}$}
{$x=\dfrac{\pi}{12}+k\pi$  với $k\in\mathbb{Z}$}
\loigiai{
Ta có $\cot 3x=-1\Leftrightarrow 3x=-\dfrac{\pi}{4}+k\pi\Leftrightarrow x=-\dfrac{\pi}{12}+k\dfrac{\pi}{3}$ với $k\in\mathbb{Z}$.
}
\end{ex}
