	Cho hàm số $ f(x) = \dfrac{x}{ (x - 1)(x - 2) \cdots (x - 2019) } $. Giá trị của $ f'(0) $ là
	\choice
	{\True $ - \dfrac{1}{2019!} $}
	{$ \dfrac{1}{2019!} $}
	{$ -2019! $}
	{$ 2019! $}
	\loigiai{
	Ta có $ f'(0) = \lim \limits_{x \to 0} \dfrac{f(x) - f(0)}{x - 0} = \lim \limits_{x \to 0} \dfrac{1}{ (x - 1)(x - 2) \cdots (x - 2019) } = \dfrac{1}{ (-1)(-2) \cdots (-2019) } = -\dfrac{1}{2019!} $.	
}
\end{ex}
