\begin{ex}%[Thi thử L2, THPT Nguyễn Trung Thiên - Hà Tĩnh, 2019]%[Nguyễn Thành Nhân,12EX7]%[1D4B5-2]
	Cho hàm số $y=\mathrm{e}^{-2x}$. Mệnh đề nào dưới đây là đúng?  
	\choice
	{$y''+y'-y=0$ }
	{$y''+y'+y=0$}
	{$y''+y'+2y=0$}
	{\True $y''+y'-2y=0$}
	\loigiai{
Ta có $y'=-2\mathrm{e}^{-2x},y''=4\mathrm{e}^{-2x}$. Do đó $y''+y'-2y=0$.  }
\end{ex}
