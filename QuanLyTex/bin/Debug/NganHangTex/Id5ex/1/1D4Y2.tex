\begin{ex}%[Thi thử lần I, Sở GD&ĐT Sơn La 2019]%[Nguyễn Anh Quốc,  dự án EX5]%[1D4Y2-7]
	$\lim\limits_{x \to - \infty} \dfrac{2x+1}{x-3}$ bằng
	\choice
	{\True $2$}
	{$- \dfrac{1}{3}$}
	{$- \dfrac{2}{3}$}
	{$1$}
	\loigiai{
		Ta có $\lim\limits_{x \to - \infty} \dfrac{2x+1}{x-3}= \lim\limits_{x \to - \infty} \dfrac{2+ \dfrac{1}{x}}{1- \dfrac{3}{x}}=2$.
	}
\end{ex}%!Cau!%

\begin{ex}%[Thi thử, Chuyên Sơn La, 2018]%[Nguyễn Thanh Tâm, 12-EX-5-2019]%[1D4Y2-7]
	$\lim \limits_{x \to -\infty} \dfrac{2x+1}{x-3}$ bằng
	\choice
	{$-\dfrac{2}{3}$}
	{$1$}
	{\True $2$}
	{$-\dfrac{1}{3}$}
	\loigiai{
		$\lim \limits_{x \to -\infty} \dfrac{2x+1}{x-3}
		=\lim \limits_{x \to -\infty} \dfrac{2+\dfrac{1}{x}}{1-\dfrac{3}{x}}=\dfrac{2}{1}=2$.	
	}
\end{ex}%!Cau!%
\begin{ex}%[Thi tập huấn, Sở GD và ĐT - Bắc Ninh, 2019]%[Nguyễn Minh Tiến, 12EX5]%[1D4Y2-2]

	{$L=1$}

\begin{ex}%[Thi thử, THPT Trần Phú - Hà Tĩnh, lần 2, 2019]%[Đỗ Đường Hiếu, 12-EX-7-2019]%[1D4Y2-7]
	$\lim\limits_{x\to +\infty}\dfrac{2x-5}{-x+3}$ bằng
	\choice
	{$-\dfrac{5}{3}$}
	{$-1$}
	{$3$}
	{\True $-2$}
	\loigiai{
		Ta có $\lim\limits_{x\to +\infty}\dfrac{2x-5}{-x+3}=\lim\limits_{x\to +\infty}\dfrac{2-\dfrac{5}{x}}{-1+\dfrac{3}{x}}=-2$.
	}
\end{ex}%!Cau!%
\begin{ex}%[Thi thử L1, Chuyên Lương Thế Vinh Đồng Nai, 2019]%[Nguyễn Tất Thu, dự án(12EX-7)]%[1D4Y2-7]
	Tìm giới hạn $L=\lim\limits_{x \to +\infty} \dfrac{3x-1}{1-2x}$.
	\choice
	{\True $L=-\dfrac{3}{2}$}
	{$L=\dfrac{3}{2}$}
	{$L=3$}
	{$L=-\dfrac{1}{2}$}
	\loigiai{
		Ta có $\lim\limits_{x \to +\infty} \dfrac{3x-1}{1-2x}=\lim\limits_{x \to +\infty} \dfrac{3-\dfrac{1}{x}}{\dfrac{1}{x}-2}=-\dfrac{3}{2}$.
	}
\end{ex}%!Cau!%
\begin{ex}%[Thi thử,Quảng Xương 1-Thanh Hóa-L3, 2019]%[Lê Quốc Hiệp, 12EX8-2019]%[1D4Y2-7]
	Tính $\lim\limits_{x\to+\infty}\dfrac{x-2}{x+3}$.
	\choice
	{$-\dfrac{2}{3}$}
	{\True $1$}
	{$2$}
	{$-3$}
	\loigiai
	{
		Ta có $\lim\limits_{x\to+\infty}\dfrac{x-2}{x+3}=\lim\limits_{x\to+\infty}\dfrac{1-\dfrac{2}{x}}{1+\dfrac{3}{x}}=1$.
	}
\end{ex}
