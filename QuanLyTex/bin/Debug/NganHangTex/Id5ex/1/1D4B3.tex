\begin{ex}%[Thi thử L1, Star Education HCM, 2019]%[Nguyễn Ngọc Dũng, dự án 12EX6]%[1D4B3-5]
Cho hàm số $f(x)=\heva{&\dfrac{2x^2-16x+30}{x-3} &\text{nếu } x\neq 3 \\
&4m &\text{nếu } x=3}$. Biết rằng $f(x)$ liên tục trên $\mathbb{R}$. Tính $f(m)$.
\choice
{$12$}
{$-2$}
{\True $-12$}
{$2$}
\loigiai{
Do $f(x)$ liên tục trên $\mathbb{R}$ nên 
$$\lim\limits_{x\to 3} f(x) = f(3) \Leftrightarrow \lim\limits_{x\to 3} \dfrac{2x^2-16x+30}{x-3} = 4m \Leftrightarrow -4 = 4m \Leftrightarrow m=-1.$$
Vậy $f(m)=f(-1) = -12$. 	
}
\end{ex}
