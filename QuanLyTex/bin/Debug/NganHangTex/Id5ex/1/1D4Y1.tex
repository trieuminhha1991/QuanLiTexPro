\begin{ex}%[Hải Phòng, 2018]%[Phan Anh Tiến, 12-EX-05]%[1D4Y1-1]
	Cho dãy số $(u_n)$ với $u_n=a\cdot 3^n$ ($a$ là hằng số). Khẳng định nào sau đây là \textbf{sai}?
	\choice
	{$ u_{n+1}=a\cdot 3^{n+1} $}
	{Với $a>0$ thì dãy tăng}
	{\True $ u_{n+1}-u_n=3\cdot a $}
	{với $a<0$ thì dãy giảm}
	\loigiai{Ta có $u_{n+1}-u_{n}=a\cdot 3^{n+1}-a\cdot 3^n=2a\cdot 3^n$, do đó $ u_{n+1}-u_n=3\cdot a $ là sai.
		}
\end{ex}
